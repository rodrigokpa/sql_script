CREATE PROCEDURE [dbo].[SP_VERIFICA_INDICE_FRAG]
	@pOBJECTNAME SYSNAME
AS
BEGIN

	SELECT	A.INDEX_ID,
			B.NAME,
			INDEX_COL(@pOBJECTNAME,A.INDEX_ID,1) INDEX_COLUMN,
			A.AVG_FRAGMENTATION_IN_PERCENT
	FROM	SYS.DM_DB_INDEX_PHYSICAL_STATS (DB_ID(),OBJECT_ID(@pOBJECTNAME),NULL,NULL,NULL) AS A
			JOIN SYS.INDEXES AS B
			ON (A.OBJECT_ID = B.OBJECT_ID AND
				A.INDEX_ID = B.INDEX_ID);

END
