CREATE FUNCTION DBO.fnMascaraValor(@VALOR_TMP MONEY)
RETURNS VARCHAR(40)
AS
BEGIN
	DECLARE @RESULTADO VARCHAR(40);
	SET @RESULTADO = REPLACE(REPLACE(LEFT(CONVERT(VARCHAR,@VALOR_TMP,1),LEN(CONVERT(VARCHAR,@VALOR_TMP,1))-3),',','.') + ','+ 
		RIGHT(CONVERT(VARCHAR,@VALOR_TMP,1),2),',00','');
	
	RETURN @RESULTADO;
END;
