CREATE PROCEDURE [dbo].[SP_VERIFICA_INDICE]
	@pOBJETO SYSNAME
AS
BEGIN

	SET NOCOUNT ON
	SET TRANSACTION ISOLATION LEVEL READ UNCOMMITTED

	SELECT	NAME AS INDEXNAME ,
			TYPE_DESC AS INDEXTYPE,
			CASE IS_DISABLED WHEN 0 THEN 'ENABLED'
				 ELSE 'DISABLED'
			END INDEXUSAGE,
			FILL_FACTOR
	FROM	SYS.INDEXES
	WHERE	OBJECT_ID = OBJECT_ID(@pOBJETO)
	ORDER BY 1,3

	SET NOCOUNT OFF

END
